function cartReducer(state, action) { switch (action.type) { case 'ADD': { const exists = state.items.find(i => i.id === action.payload.id); if (exists) return { ...state, items: state.items.map(i => i.id === action.payload.id ? { ...i, qty: i.qty + 1 } : i) }; return { ...state, items: [...state.items, { ...action.payload, qty: 1 }] }; } case 'REMOVE': return { ...state, items: state.items.filter(i => i.id !== action.payload) }; case 'INCREMENT': return { ...state, items: state.items.map(i => i.id === action.payload ? { ...i, qty: i.qty + 1 } : i) }; case 'DECREMENT': return { ...state, items: state.items.map(i => i.id === action.payload ? { ...i, qty: Math.max(1, i.qty-1) } : i) }; case 'CLEAR': return { items: [] }; default: return state; } }

The decision rule

// Ask: would you naturally describe this as an "event"? // ✅ Sounds like an event → useReducer dispatch({ type: 'SUBMIT_ORDER' }); dispatch({ type: 'APPLY_COUPON', payload: code }); dispatch({ type: 'MOVE_TO_STEP', payload: 3 }); // ✅ Simple value flip → useState is fine setIsOpen(!isOpen); setCount(n => n + 1);

Pitfall 1 — mutating state in the reducer

// ❌ Mutation — no re-render case 'ADD': state.items.push(action.payload); return state; // same reference!
// ✅ Return a new object case 'ADD': return { ...state, items: [...state.items, action.payload] };
The reducer must return a brand new object/array. React compares references — same object = no re-render.

Pitfall 2 — side effects inside the reducer

// ❌ Side effects in reducer case 'SUBMIT': fetch('/api/order', {...}); // NO localStorage.setItem(...); // NO return { ...state, submitted: true };
// ✅ Side effects in useEffect useEffect(() => { if (state.submitted) { fetch('/api/order', {...}); } }, [state.submitted]);
React Strict Mode calls reducers twice in development to enforce purity. Any side effect will run twice.

Pitfall 3 — missing default case

// ❌ Returns undefined on unknown action function reducer(state, action) { switch(action.type) { case 'A': return { ...state, a: 1 }; // no default — crashes! } }
// ✅ Always return state as fallback function reducer(state, action) { switch(action.type) { case 'A': return { ...state, a: 1 }; default: return state; // always! } }

Pitfall 4 — reading state right after dispatch

// ❌ state is still the old value right after dispatch dispatch({ type: 'INCREMENT' }); console.log(state.count); // still the OLD count! // ✅ Compute the next value yourself if you need it now const nextCount = state.count + 1; dispatch({ type: 'SET', payload: nextCount }); console.log(nextCount); // ✓
dispatch queues an update — the component re-renders first, then you'll see the new state.
